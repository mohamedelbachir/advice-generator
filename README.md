### What I learned
How to effitiency simple API

```js
async function fetchData(setData,setLoading,setError,URL){
  setLoading(true);
  setError(null);
  try{
      const res=await fetch(URL)
      const json=await res.json()
      window.setTimeout(()=>{
        setData(json)
        setLoading(false)


      },1800)
    }catch(e){
      setLoading(false)
      setError(e)
  }
}
```
